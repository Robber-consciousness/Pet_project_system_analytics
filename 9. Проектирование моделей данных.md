Физический уровень ER-d

 ![](uploads/b6d3c67f-6658-473a-afe4-34a0faf466b3/d5becc71-0701-4ad2-beac-42f9ba7a5c4a/image.png "aspect=1")


```javascript

Table SCADA {
  id int [pk, increment]
  название varchar(255) [not null]
  версия varchar(50)
  ip_адрес varchar(100) [unique]
}

Table PLC {
  id int [pk, increment]
  scada_id int [not null]
  название varchar(255) [not null]
  модель varchar(100)
  статус varchar(50)

  indexes {
    (scada_id)
  }
}

Table Датчики {
  id int [pk, increment]
  название varchar(255) [not null]
  тип varchar(100)
  единица_измерения varchar(50)
  значение decimal(10,2)
}

Table PLC_Датчики {
  plc_id int [not null]
  датчик_id int [not null]

  indexes {
    (plc_id, датчик_id) [pk]
  }
}

Ref: PLC.scada_id > SCADA.id

Ref: PLC_Датчики.plc_id > PLC.id
Ref: PLC_Датчики.датчик_id > Датчики.id
```
